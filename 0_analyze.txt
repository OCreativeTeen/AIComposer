    def finish_av_for_scenarios(self, raw_scenarios, current_scenario):
        current_id = (current_scenario["id"]/100)*100
        main_audio = get_file_path(current_scenario, "main_audio")
        main_video = get_file_path(current_scenario, "main_video")
        start_time = 0.0
        for i, s in enumerate(raw_scenarios):
            s["id"] = current_id + 100
            s["wan_style"] = current_scenario["wan_style"]
            s["wan_shot"] = current_scenario["wan_shot"]
            s["wan_angle"] = current_scenario["wan_angle"]
            s["wan_color"] = current_scenario["wan_color"]
            s["general_location"] = current_scenario["general_location"]

            s["zero"] = current_scenario["zero"]
            s["main_audio"] = main_audio
            s["main_video"] = main_video
            second = get_file_path(current_scenario, "second")
            if second:
                s["second"] = second    
            second_audio = get_file_path(current_scenario, "second_audio")
            if second_audio:
                s["second_audio"] = second_audio

            duration = s["duration"]
            if isinstance(duration, str):
                duration = float(duration)
            if i == len(raw_scenarios) - 1:
                duration = self.ffmpeg_audio_processor.get_duration(main_audio) - start_time
            s["duration"] = duration

            a = self.ffmpeg_audio_processor.audio_cut_fade(main_audio, start_time, duration)
            olda, newa =self.refresh_scenario_media(s, "clip_audio", ".wav", a)
            safe_remove(olda)

            v = self.ffmpeg_processor.resize_video(main_video, None, None, start_time, start_time + duration)
            self.refresh_scenario_media(s, "clip_video", ".mp4", v)

            start_time += duration
