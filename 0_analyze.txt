    # from clipboard, get the mp4 url, download it to temp folder, then try to replace the scene[track]
    def download_channel_media(self, track):
        """从剪贴板获取 mp4 URL，下载到临时文件夹，然后替换场景的 track"""
        try:
            # 获取剪贴板中的 URL
            url = self.root.clipboard_get()
            if not url or not url.strip():
                messagebox.showwarning("警告", "剪贴板中没有URL")
                return
            
            url = url.strip()
            
            # 验证 URL 格式
            if not (url.startswith("http://") or url.startswith("https://")):
                messagebox.showerror("错误", f"无效的URL格式: {url}")
                return
            
            # 确保临时文件夹存在
            temp_dir = config.get_temp_path(self.workflow.pid)
            os.makedirs(temp_dir, exist_ok=True)
            
            # 生成临时文件路径
            temp_path = os.path.join(temp_dir, f"{uuid.uuid4()}.mp4")
            
            # 在后台线程中下载文件
            def download_task():
                download_url = url  # 保存到局部变量
                download_path = temp_path  # 保存到局部变量
                media_track = track  # 保存到局部变量
                
                try:
                    self.root.after(0, lambda: self.log_to_output(self.video_output, f"📥 开始下载视频..."))
                    self.root.after(0, lambda u=download_url: self.log_to_output(self.video_output, f"URL: {u}"))
                    
                    # 创建 session 以保持 cookies
                    session = requests.Session()
                    
                    # 设置请求头，模拟浏览器请求以避免 403 错误
                    headers = {
                        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
                        'Accept': '*/*',
                        'Accept-Language': 'en-US,en;q=0.9',
                        'Accept-Encoding': 'gzip, deflate, br',
                        'Connection': 'keep-alive',
                        'Sec-Fetch-Dest': 'video',
                        'Sec-Fetch-Mode': 'no-cors',
                        'Sec-Fetch-Site': 'cross-site'
                    }
                    
                    # 尝试从 URL 提取域名作为 Referer
                    try:
                        from urllib.parse import urlparse
                        parsed_url = urlparse(download_url)
                        referer = f"{parsed_url.scheme}://{parsed_url.netloc}"
                        headers['Referer'] = referer
                    except:
                        pass
                    
                    # 下载文件
                    response = session.get(download_url, headers=headers, stream=True, timeout=60, allow_redirects=True)
                    
                    # 如果返回 403，尝试移除一些可能导致问题的头
                    if response.status_code == 403:
                        self.root.after(0, lambda: self.log_to_output(self.video_output, f"⚠️ 收到 403 错误，尝试简化请求头..."))
                        # 简化请求头重试
                        simple_headers = {
                            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36'
                        }
                        response = session.get(download_url, headers=simple_headers, stream=True, timeout=60, allow_redirects=True)
                    
                    response.raise_for_status()
                    
                    # 获取文件大小（如果可用）
                    total_size = int(response.headers.get('content-length', 0))
                    
                    # 写入文件
                    downloaded = 0
                    last_update_time = time.time()
                    update_interval = 0.5  # 每0.5秒更新一次进度
                    
                    with open(download_path, 'wb') as f:
                        for chunk in response.iter_content(chunk_size=8192):
                            if chunk:
                                f.write(chunk)
                                downloaded += len(chunk)
                                
                                # 限制更新频率，避免UI阻塞
                                current_time = time.time()
                                if total_size > 0 and (current_time - last_update_time) >= update_interval:
                                    percent = (downloaded / total_size) * 100
                                    # 使用默认参数避免闭包问题
                                    self.root.after(0, lambda d=downloaded, t=total_size, p=percent: self.log_to_output(
                                        self.video_output, 
                                        f"下载进度: {p:.1f}% ({d / 1024 / 1024:.1f} MB / {t / 1024 / 1024:.1f} MB)"
                                    ))
                                    last_update_time = current_time
                    
                    # 验证文件是否存在且不为空
                    if not os.path.exists(download_path) or os.path.getsize(download_path) == 0:
                        raise Exception("下载的文件为空或不存在")
                    
                    file_size = os.path.getsize(download_path) / (1024 * 1024)  # MB
                    self.root.after(0, lambda fs=file_size: self.log_to_output(
                        self.video_output, 
                        f"✅ 视频下载完成！文件大小: {fs:.1f} MB"
                    ))
                    
                    # 在主线程中处理视频替换
                    self.root.after(0, lambda p=download_path, t=media_track: self.handle_video_replacement(p, "replace", t))
                    
                except requests.exceptions.RequestException as e:
                    # 检查是否是 403 错误
                    if hasattr(e, 'response') and e.response is not None:
                        status_code = e.response.status_code
                        if status_code == 403:
                            error_msg = f"下载失败: 403 Forbidden\n\n" \
                                       f"服务器拒绝了请求。可能的原因：\n" \
                                       f"1. URL 需要登录认证或 cookies\n" \
                                       f"2. URL 已过期或需要特殊权限\n" \
                                       f"3. 服务器检测到自动化请求\n\n" \
                                       f"建议：\n" \
                                       f"- 尝试在浏览器中打开 URL 确认是否可以访问\n" \
                                       f"- 如果浏览器可以访问，可能需要手动下载文件\n" \
                                       f"- 或者使用浏览器扩展复制完整的请求头（包括 cookies）"
                        else:
                            error_msg = f"下载失败: {status_code} {str(e)}"
                    else:
                        error_msg = f"下载失败: {str(e)}"
                    
                    self.root.after(0, lambda msg=error_msg: self.log_to_output(self.video_output, f"❌ {msg}"))
                    self.root.after(0, lambda msg=error_msg: messagebox.showerror("错误", msg))
                    # 清理临时文件
                    if os.path.exists(download_path):
                        try:
                            os.remove(download_path)
                        except:
                            pass
                except Exception as e:
                    error_msg = f"处理失败: {str(e)}"
                    self.root.after(0, lambda msg=error_msg: self.log_to_output(self.video_output, f"❌ {msg}"))
                    self.root.after(0, lambda msg=error_msg: messagebox.showerror("错误", msg))
                    # 清理临时文件
                    if os.path.exists(download_path):
                        try:
                            os.remove(download_path)
                        except:
                            pass
            
            # 启动下载线程
            download_thread = threading.Thread(target=download_task, daemon=True)
            download_thread.start()
            
        except tk.TclError:
            messagebox.showwarning("警告", "无法访问剪贴板")
        except Exception as e:
            messagebox.showerror("错误", f"下载失败: {str(e)}")
